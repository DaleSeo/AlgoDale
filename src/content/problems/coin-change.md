```py
    def coinChange(self, coins: List[int], amount: int) -> int:
        min_count = amount + 1

        def dfs(count, total):
            nonlocal min_count

            if total == amount:
                min_count = min(count, min_count)
                return

            for coin in coins:
                if total + coin <= amount:
                    dfs(count + 1, total + coin)

        dfs(0, 0)

        return min_count if min_count < amount + 1 else -1
```
