  for (let i = 2; i < dp.length; i++) {
    dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i - 1]);
  }
  return dp[dp.length - 1];
}

    def rob(self, nums: List[int]) -> int:
        prev, curr = 0, 0
