```py
class Codec:
    def encode(self, strs: List[str]) -> str:
        text = ""
        for s in strs:
            text += str(len(s)) + ":" + s
        return text

    def decode(self, s: str) -> List[str]:
        strs, i = [], 0
        while i < len(s):
            delimiter = s.find(":", i)
            length = int(s[i:delimiter])
            strs.append(s[delimiter + 1 : delimiter + 1 + length])
            i = delimiter + 1 + length
        return strs
```
